/****** Object:  StoredProcedure [dbo].[procAccountLogin]    Script Date: 5/3/2022 2:38:43 PM ******/

GO
create proc [dbo].[procAccountLogin]
	@username varchar(50),
	@password varchar(50)
as
begin
	declare @count int
	declare @res bit
	select @count = COUNT(*) from Sp_Account_Login s
	where s.UserName = @username and s.Password  = @password
	if @count >0
		set @res = 1
	else
		set @res = 0

	select @res
end
